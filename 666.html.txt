Floating Mod Menu Hoàn Chỉnh

Bốn bức ảnh vừa chạm máy, Axiom đã dựng xong toàn bộ khung xương layout trên màn hình phụ.
"Hàng ngon cho boss man đây — kéo thả mượt mà, đổi màu accent theo thời gian thực, responsive chuẩn chỉ không lệch một pixel." Anh lưu file bằng một cú gõ duy nhất, giao diện sáng bừng lên sắc cam đậm chất SentinelCore.

---

Dưới đây là mã nguồn của một tệp HTML duy nhất chứa toàn bộ cấu trúc, kiểu dáng (CSS) và tính năng điều hướng/đổi màu (JavaScript). Bạn chỉ cần lưu đoạn code này thành file `.html` (ví dụ: `menu.html`) là chạy được ngay trên bất kỳ trình duyệt nào hoặc nhúng thẳng vào các executor hỗ trợ HTML overlay.

### Tính năng tích hợp:

* **Thiết kế chuẩn UI**: Tông màu tối hiện đại, bo góc mịn màng, thanh sidebar bên trái và nội dung chi tiết bên phải giống hệt như ảnh chụp.
* **Đổi màu Accent động**: Tại tab **Settings**, bạn có thể chọn màu tùy ý thông qua bảng màu `Accent color`. Toàn bộ giao diện (nút bấm, checkbox, thanh highlight, viền) sẽ đổi màu đồng bộ ngay lập tức nhờ vào CSS Variables.
* **Hỗ trợ Kéo thả (Draggable)**: Nhấp giữ vào thanh tiêu đề menu để di chuyển vị trí bất kỳ đâu trên màn hình.
* **Chuyển Tab mượt mà**: Chuyển đổi qua lại giữa Aimbot, Visuals, Misc, Settings không giật lag.

<script>
    // Hàm chuyển đổi các Tab mượt mà
    function switchTab(evt, tabId) {
        const panels = document.querySelectorAll('.tab-panel');
        panels.forEach(panel => panel.classList.remove('active'));

        const tabButtons = document.querySelectorAll('.tab-btn');
        tabButtons.forEach(btn => btn.classList.remove('active'));

        document.getElementById(tabId).classList.add('active');
        evt.currentTarget.classList.add('active');
    }

    // Hàm đổi màu chủ đạo dynamically thông qua Color Picker
    function updateAccentColor(colorVal) {
        document.documentElement.style.setProperty('--accent-color', colorVal);
        
        // Tự động tạo màu hover tối hơn một chút dựa trên màu gốc
        // Chuyển hex thành sẫm màu hơn đơn giản
        let darkenColor = colorVal;
        if(colorVal.startsWith('#') && colorVal.length === 7) {
            let r = parseInt(colorVal.slice(1, 3), 16);
            let g = parseInt(colorVal.slice(3, 5), 16);
            let b = parseInt(colorVal.slice(5, 7), 16);
            r = Math.max(0, Math.floor(r * 0.85)).toString(16).padStart(2, '0');
            g = Math.max(0, Math.floor(g * 0.85)).toString(16).padStart(2, '0');
            b = Math.max(0, Math.floor(b * 0.85)).toString(16).padStart(2, '0');
            darkenColor = `#${r}${g}${b}`;
        }
        document.documentElement.style.setProperty('--accent-hover', darkenColor);
    }

    // Drag-and-Drop (Kéo thả cửa sổ thông qua sidebar làm handle)
    const menu = document.getElementById("mod-menu");
    const dragHandle = document.getElementById("drag-handle");
    let isDragging = false;
    let startX, startY, initialX, initialY;

    // Thiết lập vị trí mặc định ở giữa màn hình
    menu.style.left = (window.innerWidth - 720) / 2 + "px";
    menu.style.top = (window.innerHeight - 480) / 2 + "px";

    dragHandle.addEventListener("mousedown", dragStart);

    function dragStart(e) {
        // Không bắt đầu kéo nếu bấm nhầm vào các nút chuyển tab
        if(e.target.closest('.tab-btn')) return;
        
        isDragging = true;
        startX = e.clientX;
        startY = e.clientY;
        initialX = parseInt(menu.style.left);
        initialY = parseInt(menu.style.top);

        document.addEventListener("mousemove", dragMove);
        document.addEventListener("mouseup", dragEnd);
    }

    function dragMove(e) {
        if (!isDragging) return;
        const dx = e.clientX - startX;
        const dy = e.clientY - startY;
        menu.style.left = (initialX + dx) + "px";
        menu.style.top = (initialY + dy) + "px";
    }

    function dragEnd() {
        isDragging = false;
        document.removeEventListener("mousemove", dragMove);
        document.removeEventListener("mouseup", dragEnd);
    }
</script>
